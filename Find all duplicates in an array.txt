class Solution {
public:
    vector<int> findDuplicates(vector<int>& nums) {
        vector<int>ans;
        unordered_map<int,int> count;
#map is like dictionaary which stores key value pair, by default value=0

        for(auto x:nums)
        count[x]++;

        for(int i=0;i<nums.size();i++)
        {
        if(count[i]==2)
        ans.push_back(i);
        }
         return ans;
    }
};
